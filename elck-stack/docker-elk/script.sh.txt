# convert CRLF -> LF (either dos2unix if installed or sed fallback)
# recommended files to fix:
files="setup/entrypoint.sh setup/lib.sh"//what is this? 
# if you have dos2unix:
dos2unix $files || true

# fallback using sed (works on Linux/mac):
for f in $files; do sed -i 's/\r$//' "$f"; done

# make sure they are executabs
chmod +x setup/entrypoint.sh setup/lib.sh
